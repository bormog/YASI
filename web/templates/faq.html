How To

Create

create table users (primary key uid, first_name, last_name, salary);

Describe

describe users;

Insert

insert into users set uid=1, first_name='Alex', last_name='Gor', salary=1000;
insert into users set uid=2, first_name='Bob', last_name='Ford', salary=200;
insert into users set uid=3, first_name='Alex', last_name='Smith', salary=3000;

Select

select 1;
select (2+3) * (10-8) / 2;
select uid, first_name, last_name from users;
select uid, first_name from users where uid=1;
select uid, first_name, salary from users where first_name='Alex', salary=1000;
select uid, last_name, salary from users order by salary desc limit 2;
