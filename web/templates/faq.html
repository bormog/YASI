How To

Create

create table users (primary key uid);
create table users (primary key uid, first_name, last_name, position);

Describe

describe users;

Insert

insert into users set uid=1;
insert into users set uid=2, first_name='Alex', last_name='Gor', position=1;

select uid from users;
select uid from users where uid=2;
select uid, first_name, position from users where first_name='Alex', position=1;
